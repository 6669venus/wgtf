CMAKE_MINIMUM_REQUIRED( VERSION 2.8.8 )
PROJECT( automation )

INCLUDE( NGTStandardProject )

SET( V0_INTERFACES
	interfaces/v0/automation_interface_v0.hpp
)
SOURCE_GROUP( "v0" FILES ${V0_INTERFACES} )


SET( ALL_SRCS
	interfaces/automation_interface.hpp
)
SOURCE_GROUP( "" FILES ${ALL_SRCS} )


BW_BLOB_SOURCES( BLOB_SRCS
	${V0_INTERFACES}
	${ALL_SRCS}
)

# Use ADD_CUSTOM_TARGET because this target has no cpp files
ADD_CUSTOM_TARGET( automation SOURCES
	${BLOB_SRCS}
)
 
BW_PROJECT_CATEGORY( automation "BigWorld Libs" )
